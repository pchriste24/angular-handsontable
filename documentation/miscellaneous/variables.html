src/shared/validatorFns.ts

emailValidator
emailValidator:
Default value : (value: string) => { if (!value || value.trim() === '') { return true; } else if (/.+@.+/.test(value)) { return true; } else { return false; } }
requiredValidator
requiredValidator:
Default value : (value: string) => { if (!value || value.trim() === '') { return false; } else { return true; } }
ssnValidator
ssnValidator:
Default value : (value: string) => { if (!value || value.trim() === '') { return true; } else if (/^((?!666|000)[0-8][0-9\_]{2}\-(?!00)[0-9\_]{2}\-(?!0000)[0-9\_]{4})*$/.test(value)) { return true; } else { return false; } }
validate
validate:
Default value : (vTypes: validatorTypes[]) => (value: string, callback) => { let validationFailed = false; for (let index = 0; index < vTypes.length; index++) { const v = vTypes[index]; if (v === validatorTypes.REQUIRED) { if (!requiredValidator(value)) { callback(false); validationFailed = true; break; } } else if (v === validatorTypes.EMAIL) { if (!emailValidator(value)) { callback(false); validationFailed = true; break; } } else if (v === validatorTypes.SSN) { if (!ssnValidator(value)) { callback(false); validationFailed = true; break; } } } if (!validationFailed) { callback(true); } }
